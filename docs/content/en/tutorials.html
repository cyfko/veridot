Tutorials

These hands-on tutorials will teach you how to build your first Veridot project, verify tokens at the gateway level, and write custom validation rules.

Tutorial 1: Building Your First Project

In this tutorial, we will set up a standard Java project structure with Maven and implement a mock authentication system using Veridot.

Tutorial 2: Verifying a Token

This tutorial shows how an API gateway microservice verifies an incoming user token. Veridot validates token structure, signature, and liveness locally without calling the issuer.

import io.github.cyfko.veridot.core.*;
import io.github.cyfko.veridot.core.impl.*;
import io.github.cyfko.veridot.core.exceptions.*;
import java.security.PublicKey;

public class GatewayVerifier {
    private final TokenVerifier verifier;

    public GatewayVerifier(Broker broker, PublicKey authorityPublicKey) {
        // Setup trust root with issuer identity check
        TrustRoot trustRoot = new PublicKeyTrustRoot() {
            @Override
            public PublicKey resolve(String issuer) {
                return authorityPublicKey;
            }
            @Override
            public boolean isRootIdentity(String issuer) {
                return "auth-authority-01".equals(issuer);
            }
        };

        // Initialize TokenVerifier
        this.verifier = new GenericSignerVerifier(broker, trustRoot);
    }

    public String handleRequest(String authHeader) {
        if (authHeader == null || !authHeader.startsWith("Bearer ")) {
            return "401 Unauthorized - Missing Token";
        }
        
        String token = authHeader.substring(7);
        try {
            // Verify and extract payload
            VerifiedData verified = verifier.verify(token, payload -> payload);
            
            // Success
            return "200 OK - Access Granted. User details: " + verified.data();
        } catch (BrokerExtractionException e) {
            // Handle stale or revoked states
            return "403 Forbidden - Token Invalid or Revoked: " + e.getMessage();
        } catch (Exception e) {
            // Cryptographic or envelope parsing failure
            return "401 Unauthorized - Verification Failed: " + e.getMessage();
        }
    }
}

Tutorial 3: Writing a Custom Rule

Rules in Veridot run during the verification pipeline. You can enforce policies like domain restrictions or minimum user levels by writing custom rules.

Create a class that implements the Rule interface. This example ensures that only users with a verified email address can access the system:

package com.example;

import io.github.cyfko.veridot.core.Rule;
import io.github.cyfko.veridot.core.exceptions.RuleValidationException;
import java.util.Map;

public class EmailVerificationRule implements Rule {
    
    @Override
    public String getName() {
        return "EMAIL_VERIFICATION_RULE";
    }

    @Override
    public void validate(Map claims) throws RuleValidationException {
        Object emailVerified = claims.get("email_verified");
        
        if (emailVerified == null) {
            throw new RuleValidationException("Email status missing from token claims");
        }
        
        if (emailVerified instanceof Boolean && !((Boolean) emailVerified)) {
            throw new RuleValidationException("Email address must be verified");
        }
        
        if (emailVerified instanceof String && !"true".equalsIgnoreCase((String) emailVerified)) {
            throw new RuleValidationException("Email address must be verified");
        }
    }
}

To register your custom rule, pass it to the verifier builder during initialization:

GenericSignerVerifier verifier = GenericSignerVerifier.builder(broker, trustRoot)
    .addRule(new EmailVerificationRule())
    .build();
