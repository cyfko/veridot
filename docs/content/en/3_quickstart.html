Quickstart Guide

This guide will help you install Veridot Core, configure a verifier instance, sign a cryptographic session token, validate it, and execute a revocation loop.

Step 1: Installation

Include the Veridot Core library in your project build file:

<dependency>
    <groupId>io.github.cyfko</groupId>
    <artifactId>veridot-core</artifactId>
    <version>4.0.0</version>
</dependency>
implementation 'io.github.cyfko:veridot-core:4.0.0'

Step 2: Implement Token Signing & Verification

Create a class named VeridotDemo.java in your project. This sample code demonstrates the complete life cycle of a session under the Veridot protocol:

import io.github.cyfko.veridot.core.*;
import io.github.cyfko.veridot.core.impl.*;
import io.github.cyfko.veridot.core.exceptions.*;
import java.security.*;

public class VeridotDemo {
    public static void main(String[] args) throws Exception {
        // 1. Initialize an InMemoryBroker for local testing
        Broker broker = new InMemoryBroker();
        
        // 2. Generate a long-term cryptographic key pair (RSA 2048) for the signer
        KeyPairGenerator keyGen = KeyPairGenerator.getInstance("RSA");
        keyGen.initialize(2048);
        KeyPair keyPair = keyGen.generateKeyPair();
        
        // 3. Define the TrustRoot to resolve the public key dynamically
        TrustRoot trustRoot = new PublicKeyTrustRoot() {
            @Override
            public PublicKey resolve(String issuer) {
                // In production, resolve this key from a KMS or public key store
                return keyPair.getPublic();
            }
            @Override
            public boolean isRootIdentity(String issuer) {
                // Identify the root publisher identity
                return "auth-authority-01".equals(issuer);
            }
        };

        // 4. Instantiate the GenericSignerVerifier service
        GenericSignerVerifier service = new GenericSignerVerifier(
            broker,
            trustRoot,
            "auth-authority-01",
            keyPair.getPrivate(),
            (byte) 0x01 // Envelope signature algorithm (SHA256withRSA)
        );

        System.out.println("Veridot service initialized successfully.");

        // 5. Sign a session for "user-99" with 10 minutes validity
        Configurer configurer = BasicConfigurer.builder()
            .groupId("user-99")
            .sequenceId("session-abc-456")
            .validity(600) // Duration in seconds (10 minutes)
            .distribution(DistributionMode.DIRECT) // Return a standard JWT
            .build();

        String token = service.sign("{\"role\":\"admin\",\"name\":\"Alice\"}", configurer);
        System.out.println("\nGenerated JWT Token:\n" + token);

        // 6. Verify the token immediately
        System.out.println("\n--- Verifying active token ---");
        try {
            VerifiedData<String> verified = service.verify(token, payload -> payload);
            System.out.println("Verification succeeded!");
            System.out.println("Payload data: " + verified.data());
            System.out.println("Is session active: " + verified.isLive());
        } catch (Exception e) {
            System.err.println("Verification failed: " + e.getMessage());
        }

        // 7. Revoke the session manually
        System.out.println("\n--- Revoking the session ---");
        service.revoke(Scope.group("user-99"), "session-abc-456");

        // 8. Try verifying again. It must fail instantly
        System.out.println("\n--- Verifying revoked token ---");
        try {
            service.verify(token, payload -> payload);
            System.err.println("CRITICAL ERROR: Token was verified after revocation!");
        } catch (BrokerExtractionException e) {
            System.out.println("Success: Token verification failed as expected. Reason: " + e.getMessage());
        }

        // Shutdown background reconciliation schedulers
        service.close();
    }
}
Compilation Requirement

Ensure your project runs on Java 17 or higher, as Veridot makes intensive use of modern Java language features (records, switch pattern matching).
